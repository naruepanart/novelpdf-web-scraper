package main

import (
	"fmt"

	"github.com/gocolly/colly/v2"
)

func main() {
	// Create a new collector
	c := colly.NewCollector()

	// Set up a callback to be called when a visited HTML element is found
	c.OnHTML("td:nth-child(3) > span", func(e *colly.HTMLElement) {
		// Extract the text and link from the HTML element
		title := e.Text
		link := e.ChildAttr("a", "href")
		// Print the results
		fmt.Printf("Title: %s\nLink: %s\n\n", title, link)
	})

	c.Visit("https://news.ycombinator.com/")
}
